20 /**
21 * Search from the MSB to LSB, looking for a 1
22 */
23 AWS_STATIC_IMPL size_t aws_clz_u32(uint32_t n) {
24 return __builtin_clzl(n);
25 }
26
27 AWS_STATIC_IMPL size_t aws_clz_i32(int32_t n) {
28 return __builtin_clz(n);
29 }
30
31 AWS_STATIC_IMPL size_t aws_clz_u64(uint64_t n) {
32 return __builtin_clzll(n);
33 }
34
35 AWS_STATIC_IMPL size_t aws_clz_i64(int64_t n) {
36 return __builtin_clzll(n);
37 }
38
39 AWS_STATIC_IMPL size_t aws_clz_size(size_t n) {
40 #if SIZE_BITS == 64
41 return aws_clz_u64(n);
42 #else
43 return aws_clz_u32(n);
44 #endif
45 }
46
47 /**

63 return __builtin_ctzll(n);
