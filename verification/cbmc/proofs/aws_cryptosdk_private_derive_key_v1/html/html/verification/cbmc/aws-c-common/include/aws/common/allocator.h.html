14 /* Allocator structure. An instance of this will be passed around for anything needing memory allocation */
15 struct aws_allocator {
16 void *(*mem_acquire)(struct aws_allocator *allocator, size_t size);
17 void (*mem_release)(struct aws_allocator *allocator, void *ptr);
18 /* Optional method; if not supported, this pointer must be NULL */
19 void *(*mem_realloc)(struct aws_allocator *allocator, void *oldptr, size_t oldsize, size_t newsize);
20 /* Optional method; if not supported, this pointer must be NULL */
21 void *(*mem_calloc)(struct aws_allocator *allocator, size_t num, size_t size);
22 void *impl;
23 };

54 /**
55 * Returns at least `size` of memory ready for usage or returns NULL on failure.
56 */
57 AWS_COMMON_API
58 void *aws_mem_acquire(struct aws_allocator *allocator, size_t size);
59
60 /**
61 * Allocates a block of memory for an array of num elements, each of them size bytes long, and initializes all its bits
62 * to zero. Returns null on failure.
63 */
64 AWS_COMMON_API
65 void *aws_mem_calloc(struct aws_allocator *allocator, size_t num, size_t size);
66
67 /**
68 * Allocates many chunks of bytes into a single block. Expects to be called with alternating void ** (dest), size_t
69 * (size). The first void ** will be set to the root of the allocation. Alignment is assumed to be sizeof(intmax_t).
70 *
71 * This is useful for allocating structs using the pimpl pattern, as you may allocate the public object and impl object
72 * in the same contiguous block of memory.
73 *
74 * Returns a pointer to the allocation.
75 */
76 AWS_COMMON_API
77 void *aws_mem_acquire_many(struct aws_allocator *allocator, size_t count, ...);
78
79 /**
80 * Releases ptr back to whatever allocated it.
81 * Nothing happens if ptr is NULL.
82 */
83 AWS_COMMON_API
84 void aws_mem_release(struct aws_allocator *allocator, void *ptr);
85
86 /*
87 * Attempts to adjust the size of the pointed-to memory buffer from oldsize to
88 * newsize. The pointer (*ptr) may be changed if the memory needs to be
89 * reallocated.
90 *
91 * If reallocation fails, *ptr is unchanged, and this method raises an
92 * AWS_ERROR_OOM error.
93 */
94 AWS_COMMON_API
95 int aws_mem_realloc(struct aws_allocator *allocator, void **ptr, size_t oldsize, size_t newsize);
96 /*
97 * Maintainer note: The above function doesn't return the pointer (as with
98 * standard C realloc) as this pattern becomes error-prone when OOMs occur.
99 * In particular, we want to avoid losing the old pointer when an OOM condition
100 * occurs, so we prefer to take the old pointer as an in/out reference argument
101 * that we can leave unchanged on failure.
102 */
103
104 enum aws_mem_trace_level {
105 AWS_MEMTRACE_NONE = 0, /* no tracing */
106 AWS_MEMTRACE_BYTES = 1, /* just track allocation sizes and total allocated */
107 AWS_MEMTRACE_STACKS = 2, /* capture callstacks for each allocation */
108 };
109
110 /*
111 * Wraps an allocator and tracks all external allocations. If aws_mem_trace_dump() is called
112 * and there are still allocations active, they will be reported to the aws_logger at TRACE level.
113 * allocator - The allocator to wrap
114 * deprecated - Deprecated arg, ignored.
115 * level - The level to track allocations at
116 * frames_per_stack is how many frames to store per callstack if AWS_MEMTRACE_STACKS is in use,
117 * otherwise it is ignored. 8 tends to be a pretty good number balancing storage space vs useful stacks.
118 * Returns the tracer allocator, which should be used for all allocations that should be tracked.
119 */
120 AWS_COMMON_API
121 struct aws_allocator *aws_mem_tracer_new(
122 struct aws_allocator *allocator,
123 struct aws_allocator *deprecated,
124 enum aws_mem_trace_level level,
125 size_t frames_per_stack);
126
127 /*
128 * Unwraps the traced allocator and cleans up the tracer.
129 * Returns the original allocator
130 */
131 AWS_COMMON_API
132 struct aws_allocator *aws_mem_tracer_destroy(struct aws_allocator *trace_allocator);
133
134 /*
135 * If there are outstanding allocations, dumps them to log, along with any information gathered
136 * based on the trace level set when aws_mem_trace() was called.
137 * Should be passed the tracer allocator returned from aws_mem_trace().
138 */
139 AWS_COMMON_API
140 void aws_mem_tracer_dump(struct aws_allocator *trace_allocator);

142 /*

145 AWS_COMMON_API

154 /*
155 * Creates a new Small Block Allocator which fronts the supplied parent allocator. The SBA will intercept
156 * and handle small allocs, and will forward anything larger to the parent allocator.
157 * If multi_threaded is true, the internal allocator will protect its internal data structures with a mutex
158 */
159 AWS_COMMON_API

162 /*
163 * Destroys a Small Block Allocator instance and frees its memory to the parent allocator. The parent
164 * allocator will otherwise be unaffected.
165 */
166 AWS_COMMON_API
