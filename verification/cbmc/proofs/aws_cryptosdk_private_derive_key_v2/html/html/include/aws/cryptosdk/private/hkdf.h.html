16 #ifndef AWS_CRYPTOSDK_PRIVATE_HKDF_H
17 #define AWS_CRYPTOSDK_PRIVATE_HKDF_H
18
19 #include <aws/common/byte_buf.h>
20
21 enum aws_cryptosdk_sha_version {
22 AWS_CRYPTOSDK_NOSHA,
23 AWS_CRYPTOSDK_SHA256,
24 AWS_CRYPTOSDK_SHA384,
25 AWS_CRYPTOSDK_SHA512
26 };
27
28 /*
29 * This function performs the HKDF extract then expand steps as described in
30 * RFC-5869. The length of the okm (output keying material) is required to be
31 * set by the user ahead of time and must be less than or equal to 255*HashLen.
32 */
33 int aws_cryptosdk_hkdf(
34 struct aws_byte_buf *okm,
35 enum aws_cryptosdk_sha_version which_sha,
36 const struct aws_byte_buf *salt,
37 const struct aws_byte_buf *ikm,
38 const struct aws_byte_buf *info);
39
40 #endif // AWS_CRYPTOSDK_PRIVATE_HKDF_H
