16 #ifndef AWS_CRYPTOSDK_EDK_H
17 #define AWS_CRYPTOSDK_EDK_H
18
19 #include <aws/common/array_list.h>
20 #include <aws/common/byte_buf.h>
21
22 #include <aws/cryptosdk/exports.h>
23
24 /*
25 * This public interface to the encrypted data key (EDK) objects is provided for
26 * developers of CMMs and keyrings only. If you are a user of the AWS Encryption
27 * SDK for C and you are not developing your own CMMs and/or keyrings, you do not
28 * need to use it and you should not do so.
29 */
30
31 /**
32 * @addtogroup cmm_kr_lowlevel
33 * @{
34 */
35
36 /**
37 * A structure representing a single encrypted data key.
38 */
39 struct aws_cryptosdk_edk {
40 struct aws_byte_buf provider_id;
41 struct aws_byte_buf provider_info;
42 struct aws_byte_buf ciphertext;
43 };
44
45 #ifdef __cplusplus
46 extern "C" {
47 #endif
48
49 /**
50 * Evaluates the set of properties that define the shape of all valid aws_cryptosdk_edk structures.
51 * It is also a cheap check, in the sense it run in constant time (i.e., no loops or recursion).
52 */
53 AWS_CRYPTOSDK_API
54 bool aws_cryptosdk_edk_is_valid(const struct aws_cryptosdk_edk *const edk);
55
56 /**
57 * Evaluates the set of properties that define the shape of a valid edk_list.
58 * if (AWS_DEEP_CHECKS == 1), loops through all elements of the list;
59 * otherwise, it completes in constant time.
60 */
61 AWS_CRYPTOSDK_API
62 bool aws_cryptosdk_edk_list_is_valid(const struct aws_array_list *edk_list);
63
64 /**
65 * Evaluates the set of properties that define the shape of a valid empty edk_list.
66 */
67 AWS_CRYPTOSDK_API
68 bool aws_cryptosdk_empty_edk_list_is_valid(const struct aws_array_list *edk_list);
69
70 /**
71 * Evaluates the set of properties that define the shape of a valid edk_list.
72 * Takes time O(list->length) as it loops through all elements of the list;
73 */
74 bool aws_cryptosdk_edk_list_elements_are_valid(const struct aws_array_list *edk_list);
75
76 /**
77 * Deallocates all memory associated with an EDK. Setting all bytes of an EDK to
78 * zero when you declare it will make this safe to call even if some buffers are unused.
79 */
80 AWS_CRYPTOSDK_API
81 void aws_cryptosdk_edk_clean_up(struct aws_cryptosdk_edk *edk);
82
83 /**
84 * Allocates an empty list of EDKs.
85 */
86 AWS_CRYPTOSDK_API
87 int aws_cryptosdk_edk_list_init(struct aws_allocator *alloc, struct aws_array_list *edk_list);
88
89 /**
90 * Deallocates all memory associated with all EDKs in the list and then deallocates the list.
91 */
92 AWS_CRYPTOSDK_API
93 void aws_cryptosdk_edk_list_clean_up(struct aws_array_list *edk_list);
94
95 /**
96 * Deallocates all memory associated with all EDKs in the list and then clears the list.
97 * The array list itself remains allocated but empty.
98 */
99 AWS_CRYPTOSDK_API
100 void aws_cryptosdk_edk_list_clear(struct aws_array_list *edk_list);
101
102 /**
103 * Copies the EDK data in src to dest.
104 */
105 AWS_CRYPTOSDK_API
106 int aws_cryptosdk_edk_init_clone(
107 struct aws_allocator *alloc, struct aws_cryptosdk_edk *dest, const struct aws_cryptosdk_edk *src);
108
109 /**
110 * Returns true if the contents of all EDK byte buffers are identical, false otherwise.
111 */
112 AWS_CRYPTOSDK_STATIC_INLINE bool aws_cryptosdk_edk_eq(
113 const struct aws_cryptosdk_edk *a, const struct aws_cryptosdk_edk *b) {
114 return aws_byte_buf_eq(&a->ciphertext, &b->ciphertext) && aws_byte_buf_eq(&a->provider_info, &b->provider_info) &&
115 aws_byte_buf_eq(&a->provider_id, &b->provider_id);
116 }
117
118 #ifdef __cplusplus
119 }
120 #endif
