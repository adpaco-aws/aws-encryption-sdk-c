20 /**
21 * Controls what log calls pass through the logger and what log calls get filtered out.
22 * If a log level has a value of X, then all log calls using a level <= X will appear, while
23 * those using a value > X will not occur.
24 *
25 * You can filter both dynamically (by setting the log level on the logger object) or statically
26 * (by defining AWS_STATIC_LOG_LEVEL to be an appropriate integer module-wide). Statically filtered
27 * log calls will be completely compiled out but require a rebuild if you want to get more detail
28 * about what's happening.
29 */
30 enum aws_log_level {
31 AWS_LL_NONE = AWS_LOG_LEVEL_NONE,
32 AWS_LL_FATAL = AWS_LOG_LEVEL_FATAL,
33 AWS_LL_ERROR = AWS_LOG_LEVEL_ERROR,
34 AWS_LL_WARN = AWS_LOG_LEVEL_WARN,
35 AWS_LL_INFO = AWS_LOG_LEVEL_INFO,
36 AWS_LL_DEBUG = AWS_LOG_LEVEL_DEBUG,
37 AWS_LL_TRACE = AWS_LOG_LEVEL_TRACE,
38
39 AWS_LL_COUNT
40 };
41
42 /**
43 * Log subject is a way of designating the topic of logging.
44 *
45 * The general idea is to support a finer-grained approach to log level control. The primary use case
46 * is for situations that require more detailed logging within a specific domain, where enabling that detail
47 * globally leads to an untenable flood of information.
48 *
49 * For example, enable TRACE logging for tls-related log statements (handshake binary payloads), but
50 * only WARN logging everywhere else (because http payloads would blow up the log files).
51 *
52 * Log subject is an enum similar to aws error: each library has its own value-space and someone is
53 * responsible for registering the value <-> string connections.
54 */
55 typedef uint32_t aws_log_subject_t;
56
57 /* Each library gets space for 2^^10 log subject entries */
58 #define AWS_LOG_SUBJECT_STRIDE_BITS 10
59 #define AWS_LOG_SUBJECT_STRIDE (1U << AWS_LOG_SUBJECT_STRIDE_BITS)
60 #define AWS_LOG_SUBJECT_BEGIN_RANGE(x) ((x)*AWS_LOG_SUBJECT_STRIDE)
61 #define AWS_LOG_SUBJECT_END_RANGE(x) (((x) + 1) * AWS_LOG_SUBJECT_STRIDE - 1)
62

91 /**
92 * We separate the log level function from the log call itself so that we can do the filter check in the macros (see
93 * below)
94 *
95 * By doing so, we make it so that the variadic format arguments are not even evaluated if the filter check does not
96 * succeed.
97 */
98 struct aws_logger_vtable {
99 int (*const log)(
100 struct aws_logger *logger,
101 enum aws_log_level log_level,
102 aws_log_subject_t subject,
103 const char *format,
104 ...)
105 #if defined(__clang__) || defined(__GNUC__) || defined(__GNUG__)
106 __attribute__((format(printf, 4, 5)))
107 #endif /* non-ms compilers: TODO - find out what versions format support was added in */
108 ;
109 enum aws_log_level (*const get_log_level)(struct aws_logger *logger, aws_log_subject_t subject);
110 void (*const clean_up)(struct aws_logger *logger);
111 };
112
113 struct aws_logger {
114 struct aws_logger_vtable *vtable;
115 struct aws_allocator *allocator;
116 void *p_impl;
117 };

119 /**
120 * The base formatted logging macro that all other formatted logging macros resolve to.
121 * Checks for a logger and filters based on log level.
122 *
123 */
124 #define AWS_LOGF(log_level, subject, ...) \
125 { \
126 AWS_ASSERT(log_level > 0); \
127 struct aws_logger *logger = aws_logger_get(); \
128 if (logger != NULL && logger->vtable->get_log_level(logger, (subject)) >= (log_level)) { \
129 logger->vtable->log(logger, log_level, subject, __VA_ARGS__); \
130 } \
131 }
132
133 /**
134 * LOGF_<level> variants for each level. These are what should be used directly to do all logging.
135 *
136 * i.e.
137 *
138 * LOGF_FATAL("Device \"%s\" not found", device->name);
139 *
140 *
141 * Later we will likely expose Subject-aware variants
142 */
143 #if !defined(AWS_STATIC_LOG_LEVEL) || (AWS_STATIC_LOG_LEVEL >= AWS_LOG_LEVEL_FATAL)
144 # define AWS_LOGF_FATAL(subject, ...) AWS_LOGF(AWS_LL_FATAL, subject, __VA_ARGS__)
145 #else
146 # define AWS_LOGF_FATAL(subject, ...)
147 #endif
148
149 #if !defined(AWS_STATIC_LOG_LEVEL) || (AWS_STATIC_LOG_LEVEL >= AWS_LOG_LEVEL_ERROR)
150 # define AWS_LOGF_ERROR(subject, ...) AWS_LOGF(AWS_LL_ERROR, subject, __VA_ARGS__)
151 #else
152 # define AWS_LOGF_ERROR(subject, ...)
153 #endif
154
155 #if !defined(AWS_STATIC_LOG_LEVEL) || (AWS_STATIC_LOG_LEVEL >= AWS_LOG_LEVEL_WARN)
156 # define AWS_LOGF_WARN(subject, ...) AWS_LOGF(AWS_LL_WARN, subject, __VA_ARGS__)
157 #else
158 # define AWS_LOGF_WARN(subject, ...)
159 #endif
160
161 #if !defined(AWS_STATIC_LOG_LEVEL) || (AWS_STATIC_LOG_LEVEL >= AWS_LOG_LEVEL_INFO)
162 # define AWS_LOGF_INFO(subject, ...) AWS_LOGF(AWS_LL_INFO, subject, __VA_ARGS__)
163 #else
164 # define AWS_LOGF_INFO(subject, ...)
165 #endif
166
167 #if !defined(AWS_STATIC_LOG_LEVEL) || (AWS_STATIC_LOG_LEVEL >= AWS_LOG_LEVEL_DEBUG)
168 # define AWS_LOGF_DEBUG(subject, ...) AWS_LOGF(AWS_LL_DEBUG, subject, __VA_ARGS__)
169 #else
170 # define AWS_LOGF_DEBUG(subject, ...)
171 #endif
172
173 #if !defined(AWS_STATIC_LOG_LEVEL) || (AWS_STATIC_LOG_LEVEL >= AWS_LOG_LEVEL_TRACE)
174 # define AWS_LOGF_TRACE(subject, ...) AWS_LOGF(AWS_LL_TRACE, subject, __VA_ARGS__)
175 #else
176 # define AWS_LOGF_TRACE(subject, ...)
177 #endif

179 /*
180 * Standard logger implementation composing three sub-components:
181 *
182 * The formatter takes var args input from the user and produces a formatted log line
183 * The writer takes a formatted log line and outputs it somewhere
184 * The channel is the transport between the two
185 */
186 struct aws_logger_pipeline {
187 struct aws_log_formatter *formatter;
188 struct aws_log_channel *channel;
189 struct aws_log_writer *writer;
190 struct aws_allocator *allocator;
191 enum aws_log_level level;
192 };
193
194 /**
195 * Options for aws_logger_init_standard().
196 * Set `filename` to open a file for logging and close it when the logger cleans up.
197 * Set `file` to use a file that is already open, such as `stderr` or `stdout`.
198 */
199 struct aws_logger_standard_options {
200 enum aws_log_level level;
201 const char *filename;
202 FILE *file;
203 };
